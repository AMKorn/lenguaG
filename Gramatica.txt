BODY -> DECLARATIONS MAIN.

MAIN -> main L_PAREN ARGS R_PAREN L_KEY INSTRUCTIONS R_KEY.

DECLARATIONS -> DECLARATION DECLARATIONS 
    | FUNCTION DECLARATIONS 
    | .

DECLARATION -> CONSTANT TYPE IDENTIFIER INITIALIZATION ENDLINE.
INITIALIZATION -> EQUAL VALUE
    | .

CONSTANT -> final
    | .

FUNCTION -> TYPE IDENTIFIER L_PAREN ARGS R_PAREN L_KEY INSTRUCTIONS R_KEY.

ARGS -> ARG MORE_ARGS
    | .

MORE_ARGS -> COMMA ARG
    | .

INSTRUCTIONS -> INSTRUCTION INSTRUCTIONS
    | .

INSTRUCTION -> DECLARATION
    | ASSIGNATION
    | SWAP
    | IF 
    | LOOP
    | FUNCTION_CALL

ASSIGNATION -> IDENTIFIER EQUAL VALUE ENDLINE

VALUE -> IDENTIFIER
    | OPERATION

OPERATION -> ADDITION
    | SUBSTRACTION
    | PRODUCT
    | DIVISION
    | MODULE
    | OP_REL.

ADDITION -> IDENTIFIER + IDENTIFIER.

SUBSTRACTION -> IDENTIFIER - IDENTIFIER

PRODUCT -> IDENTIFIER * IDENTIFIER.

DIVISION -> IDENTIFIER / IDENTIFIER

MODULE -> IDENTIFIER % IDENTIFIER

OP_REL -> OR
    | AND
    | EQUALS
    | BIGGER
    | BEQ
    | LESSER
    | LEQ
    | NEQ

OR -> IDENTIFIER || IDENTIFIER.

AND -> IDENTIFIER && IDENTIFIER

EQUALS -> IDENTIFIER == IDENTIFIER.

BIGGER -> IDENTIFIER > IDENTIFIER.

BEQ -> IDENTIFIER >= IDENTIFIER.

LESSER -> IDENTIFIER < IDENTIFIER.

LEQ -> IDENTIFIER <= IDENTIFIER.

NEQ -> IDENTIFIER != IDENTIFIER.
