BODY -> DECLARATIONS MAIN.

MAIN -> main L_PAREN ARGS R_PAREN L_KEY INSTRUCTIONS R_KEY.

DECLARATIONS -> DECLARATION DECLARATIONS 
    | FUNCTION DECLARATIONS 
    | .

DECLARATION -> CONSTANT TYPE IDENTIFIER INITIALIZATION ENDLINE.
INITIALIZATION -> EQUAL VALUE
    | .

CONSTANT -> final
    | var .

FUNCTION -> TYPE IDENTIFIER L_PAREN ARGS R_PAREN L_KEY INSTRUCTIONS R_KEY.

ARGS -> ARG MORE_ARGS
    | .

MORE_ARGS -> COMMA ARG
    | .

INSTRUCTIONS -> INSTRUCTION INSTRUCTIONS
    | .

INSTRUCTION -> DECLARATION
    | ASSIGNATION
    | SWAP
    | IF 
    | LOOP
	| FOR
    | FUNCTION_CALL.

ASSIGNATION -> IDENTIFIER EQUAL VALUE ENDLINE.

VALUE -> IDENTIFIER OPERATION.

OPERATION -> ADDITION
    | SUBSTRACTION
    | PRODUCT
    | DIVISION
    | MODULE
    | OP_REL
	| .

ADDITION -> + IDENTIFIER.

SUBSTRACTION -> - IDENTIFIER.

PRODUCT -> * IDENTIFIER.

DIVISION -> / IDENTIFIER.

MODULE -> % IDENTIFIER.

OP_REL -> OR
    | AND
    | EQUALS
    | BIGGER
    | BEQ
    | LESSER
    | LEQ
    | NEQ.

OR -> or IDENTIFIER.

AND -> && IDENTIFIER.

EQUALS -> == IDENTIFIER.

BIGGER -> > IDENTIFIER.

BEQ -> >= IDENTIFIER.

LESSER -> < IDENTIFIER.

LEQ -> <= IDENTIFIER.

NEQ -> != IDENTIFIER.

SWAP -> <<>> IDENTIFIER.

IDENTIFIER -> VARIABLE
	| FIXED_VALUE
	| FUNCTION_CALL.

VARIABLE -> [a-zA-Z_][a-zA-Z0-9_]+ ARRAY_SUFFIX.
ARRAY_SUFFIX -> '['IDENTIFIER']'.

FIXED_VALUE -> INTEGER
	| CHARACTER
	| LIST
	| BOOLEAN.

TYPE -> TYPE_INTEGER
	| TYPE_CHARACTER
	| TYPE_BOOLEAN
	| TYPE_ARRAY.
	
TYPE_INTEGER -> int.
TYPE_CHARACTER -> char.
TYPE_BOOLEAN -> boolean.
TYPE_ARRAY -> TYPE[].


IF -> if L_PAREN IDENTIFIER R_PAREN L_KEY INSTRUCTIONS R_KEY ELSE.
ELSE -> else if PAREN IDENTIFIER R_PAREN L_KEY INSTRUCTIONS R_KEY ELSE
	| else L_KEY INSTRUCTIONS R_KEY ELSE.

LOOP -> while L_PAREN IDENTIFIER R_PAREN L_KEY INSTRUCTIONS R_KEY.

FOR -> for L_PAREN IDENTIFIER INITIALIZATION ENDLINE IDENTIFIER ENDLINE INSTRUCTIONS R_PAREN L_KEY INSTRUCTIONS R_KEY.

FUNCTION_CALL -> IDENTIFIER L_PAREN ARGS R_PAREN.
